def binary_digits(decimalnumber):
    binaryvalue=[]
    if decimalnumber==0:
        return '0'
    while decimalnumber>0:
        reminder=decimalnumber%2
        binaryvalue.append(str(reminder))
        decimalnumber = decimalnumber//2
    binary_representation = ''.join(binaryvalue[::-1])
    return binary_representation
        

res=binary_digits(42)
print (res)
